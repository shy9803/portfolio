function renderProjectList(page = 1, filter = 'all') {
  prj_list_ul.innerHTML = ''; // 기존 리스트 초기화

  const filteredProjects = project.filter(prj => {
    const type = normal_skill(prj.skill);
    return filter === 'all' || type.includes(filter);
  });

  const start = (page - 1) * ITEMS_PER_PAGE;
  const end = start + ITEMS_PER_PAGE;
  const paginatedProjects = filteredProjects.slice(start, end);

  paginatedProjects.forEach(prj => {
    const li = document.createElement('li');
    const pic = document.createElement('picture');
    const img = document.createElement('img');

    img.src = './images/project_img/' + prj.img[0].main;
    img.alt = prj.name;
    img.dataset.type = normal_skill(prj.skill);

    pic.appendChild(img);
    li.appendChild(pic);
    prj_list_ul.appendChild(li);
  });

  setlist(); // 클릭 이벤트 연결
  renderPagination(filteredProjects.length, page, filter);
}

function renderPagination(totalItems, currentPage, filter) {
  const totalPages = Math.ceil(totalItems / ITEMS_PER_PAGE);
  const paginationContainer = document.getElementById('pagination');

  paginationContainer.innerHTML = '';

  for (let i = 1; i <= totalPages; i++) {
    const btn = document.createElement('button');
    btn.textContent = i;
    if (i === currentPage) btn.classList.add('active');

    btn.addEventListener('click', () => {
      renderProjectList(i, filter);
    });

    paginationContainer.appendChild(btn);
  }
}
